26       private void inicializaMatrizVerificacao() {

30                   matrizVerificacao[linha][coluna] =

33           }

36       private boolean verificaGanhador() {
37   
38           boolean ganhador = false;
39           //horizontal
40           if (matrizVerificacao[0][0].equals(matrizVerificacao[0][1]) &&
41                   matrizVerificacao[0][0].equals(matrizVerificacao[0][2])) {
42               ganhador = true;
43           }
44           if (matrizVerificacao[1][0].equals(matrizVerificacao[1][1]) &&
45                   matrizVerificacao[1][0].equals(matrizVerificacao[1][2])) {
46               ganhador = true;
47           }
48           if (matrizVerificacao[2][0].equals(matrizVerificacao[2][1]) &&
49                   matrizVerificacao[2][0].equals(matrizVerificacao[2][2])) {
50               ganhador = true;
51           }
52           //vertical
53           if (matrizVerificacao[0][0].equals(matrizVerificacao[1][0]) &&
54                   matrizVerificacao[0][0].equals(matrizVerificacao[2][0])) {
55               ganhador = true;
56           }
57           if (matrizVerificacao[0][1].equals(matrizVerificacao[1][1]) &&
58                   matrizVerificacao[0][1].equals(matrizVerificacao[2][1])) {
59               ganhador = true;
60           }
61           if (matrizVerificacao[0][2].equals(matrizVerificacao[1][2]) &&
62                   matrizVerificacao[0][2].equals(matrizVerificacao[2][2])) {
63               ganhador = true;
64           }
65           //diagonal
66           if (matrizVerificacao[0][0].equals(matrizVerificacao[1][1]) &&
67                   matrizVerificacao[0][0].equals(matrizVerificacao[2][2])) {
68               ganhador = true;
69           }
70           if (matrizVerificacao[0][2].equals(matrizVerificacao[1][1]) &&
71                   matrizVerificacao[0][2].equals(matrizVerificacao[2][0])) {
72               ganhador = true;
73           }
74           return ganhador;
75       }
